Player Movement Controller

Smooth character movement with jump mechanics

Unity C# 
using UnityEngine;

public class PlayerController : MonoBehaviour
{
    [SerializeField] private float moveSpeed = 5f;
    [SerializeField] private float jumpForce = 7f;
    private Rigidbody rb;
    private bool isGrounded;

    private void Start() => rb = GetComponent<Rigidbody>();

    private void Update()
    {
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");
        
        Vector3 movement = new Vector3(horizontal, 0, vertical) * moveSpeed * Time.deltaTime;
        transform.Translate(movement);
        
        if (Input.GetButtonDown("Jump") && isGrounded)
        {
            rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
        }
    }

    private void OnCollisionStay(Collision collision)
    {
        if (collision.gameObject.CompareTag("Ground")) isGrounded = true;
    }

    private void OnCollisionExit(Collision collision)
    {
        if (collision.gameObject.CompareTag("Ground")) isGrounded = false;
    }
}

Damage Calculation System

Modular damage handling with resistance types

Unreal Engine C++ 
#include "DamageSystem.h"
#include "Components/HealthComponent.h"

float UDamageSystem::CalculateFinalDamage(float BaseDamage, const FDamageParams& Params)
{
    float finalDamage = BaseDamage;
    
    // Apply resistance based on damage type
    if (Params.DamageType == EDamageType::Fire)
    {
        finalDamage *= (1.0f - Params.TargetFireResistance);
    }
    else if (Params.DamageType == EDamageType::Ice)
    {
        finalDamage *= (1.0f - Params.TargetIceResistance);
    }
    
    // Critical hit calculation
    if (FMath::FRand() < Params.CritChance)
    {
        finalDamage *= Params.CritMultiplier;
    }
    
    // Armor mitigation
    finalDamage *= FMath::Max(0.0f, 1.0f - (Params.TargetArmor / 100.0f));
    
    return FMath::RoundToFloat(finalDamage);
}

void UDamageSystem::ApplyDamage(AActor* Target, float DamageAmount, TSubclassOf<UDamageType> DamageType)
{
    if (UHealthComponent* HealthComp = Target->FindComponentByClass<UHealthComponent>())
    {
        HealthComp->TakeDamage(DamageAmount, FDamageEvent(DamageType), nullptr, nullptr);
    }
}

Enemy AI Behavior

State machine for patrol and chase behaviors

Godot GDScript 
extends CharacterBody2D

enum STATE { PATROL, CHASE, ATTACK }
var current_state = STATE.PATROL
var patrol_points = []
var current_patrol_index = 0
var player = null
var speed = 150
var detection_radius = 200

func _ready():
    player = get_tree().get_first_node_in_group("player")
    $DetectionArea.body_entered.connect(_on_detection_area_body_entered)
    $DetectionArea.body_exited.connect(_on_detection_area_body_exited)

func _physics_process(delta):
    match current_state:
        STATE.PATROL:
            _patrol_behavior(delta)
        STATE.CHASE:
            _chase_behavior(delta)
        STATE.ATTACK:
            _attack_behavior(delta)

func _patrol_behavior(delta):
    if patrol_points.size() == 0:
        return
        
    var target_point = patrol_points[current_patrol_index]
    var direction = (target_point - position).normalized()
    
    velocity = direction * speed
    move_and_slide()
    
    if position.distance_to(target_point) < 10:
        current_patrol_index = (current_patrol_index + 1) % patrol_points.size()

func _chase_behavior(delta):
    if player:
        var direction = (player.position - position).normalized()
        velocity = direction * speed * 1.5
        move_and_slide()

func _on_detection_area_body_entered(body):
    if body.is_in_group("player"):
        current_state = STATE.CHASE

func _on_detection_area_body_exited(body):
    if body.is_in_group("player") and get_overlapping_bodies().size() == 0:
        current_state = STATE.PATROL
